001    package com.thetransactioncompany.jsonrpc2.client;
002    
003    
004    import java.io.BufferedReader;
005    import java.io.InputStream;
006    import java.io.InputStreamReader;
007    import java.io.IOException;
008    
009    import java.net.HttpURLConnection;
010    
011    import java.util.List;
012    import java.util.Map;
013    
014    import java.util.zip.GZIPInputStream;
015    import java.util.zip.Inflater;
016    import java.util.zip.InflaterInputStream;
017    
018    
019    /**
020     * Represents the raw HTTP response to a JSON-RPC 2.0 request or notification. 
021     * Can be used to retrieve the unparsed response content and headers.
022     *
023     * @since 1.6
024     * @author Vladimir Dzhuvinov
025     * @version 1.14 (2013-03-29)
026     */
027    public class RawResponse {
028    
029    
030            /**
031             * The HTTP response headers.
032             */
033            private Map<String,List<String>> headers;
034            
035            
036            /**
037             * The HTTP response code.
038             */
039            private int statusCode;
040            
041            
042            /**
043             * The HTTP response message.
044             */
045            private String statusMessage;
046    
047    
048            /**
049             * The content length.
050             */
051            private int contentLength;
052    
053    
054            /**
055             * The content type.
056             */
057            private String contentType;
058    
059    
060            /**
061             * The content encoding.
062             */
063            private String contentEncoding;
064    
065    
066            /**
067             * The raw HTTP response content.
068             */
069            private String content;
070    
071    
072            /**
073             * No public instantiation.
074             */
075            private RawResponse() {}
076    
077    
078            /**
079             * Parses the raw HTTP response from the specified URL connection.
080             * 
081             * @param connection The URL connection, must be in state completed and 
082             *                   not {@code null}.
083             * 
084             * @throws IOException If the response content couldn't be read.
085             */
086            protected static RawResponse parse(final HttpURLConnection connection)
087                    throws IOException {
088    
089                    // Check for HTTP compression
090                    String encoding = connection.getContentEncoding();
091    
092                    InputStream is;
093    
094                    if (encoding != null && encoding.equalsIgnoreCase("gzip"))
095                            is = new GZIPInputStream(connection.getInputStream());
096    
097                    else if (encoding != null && encoding.equalsIgnoreCase("deflate"))
098                            is = new InflaterInputStream(connection.getInputStream(), new Inflater(true));
099    
100                    else
101                            is = connection.getInputStream();
102    
103    
104                    // Read the response content
105                    StringBuilder responseText = new StringBuilder();
106    
107                    BufferedReader input = new BufferedReader(new InputStreamReader(is, "UTF-8"));
108    
109                    String line;
110    
111                    while ((line = input.readLine()) != null) {
112                            responseText.append(line);
113                            responseText.append(System.getProperty("line.separator"));
114                    }
115    
116                    input.close();
117    
118                    RawResponse response = new RawResponse();
119    
120                    response.content = responseText.toString();
121                    
122                    // Save HTTP code + message
123                    response.statusCode = connection.getResponseCode();
124                    response.statusMessage = connection.getResponseMessage();
125    
126                    // Save headers
127                    response.headers = connection.getHeaderFields();
128    
129                    response.contentLength = connection.getContentLength();
130                    response.contentType = connection.getContentType();
131                    response.contentEncoding = encoding;
132    
133                    return response;
134            }
135            
136            
137            /**
138             * Gets the status code from the HTTP response message, e.g. 200 on 
139             * success.
140             * 
141             * @return The HTTP status code, or -1.
142             */
143            public int getStatusCode() {
144                    
145                    return statusCode;
146            }
147            
148            
149            /**
150             * Gets the HTTP response status message, if any, returned along with 
151             * the status code from a server.
152             *  
153             * @return The HTTP status message, or {@code null}.
154             */
155            public String getStatusMessage() {
156                    
157                    return statusMessage;
158            }
159    
160    
161            /**
162             * Returns the raw response content.
163             * 
164             * @return The raw content.
165             */
166            public String getContent() {
167    
168                    return content;
169            }
170            
171            
172            /**
173             * Returns an unmodifiable Map of the header fields. The Map keys are 
174             * Strings that represent the response header field names. Each Map 
175             * value is an unmodifiable List of Strings that represents the 
176             * corresponding field values.
177             * 
178             * @return A Map of the header fields.
179             */
180            public Map<String,List<String>> getHeaderFields() {
181                    
182                    return headers;
183            }
184    
185    
186            /**
187             * Returns the value of the named header field. If called on a 
188             * connection that sets the same header multiple times with possibly 
189             * different values, only the last value is returned.
190             *  
191             * @param name The name of the header.
192             * 
193             * @return The value of the named header field, {@code null} if there 
194             *         is no such field in the header.
195             */
196            public String getHeaderField(final String name) {
197    
198                    List <String> values = headers.get(name);
199    
200                    if (values == null | values.size() <= 0)
201                            return null;
202    
203                    return values.get(0);
204            }
205    
206    
207            /**
208             * Returns the value of the "Content-Length" header field. 
209             * 
210             * @return The content length of the response, or -1 if the content 
211             *         length  is not known.
212             */
213            public int getContentLength() {
214    
215                    return contentLength;
216            }
217    
218    
219            /**
220             * Returns the value of the "Content-Type" header field.
221             * 
222             * @return The content type of the response, or {@code null} if not 
223             *         known.
224             */
225            public String getContentType() {
226    
227                    return contentType;
228            }
229    
230            /**
231             * Returns the value of the "Content-Encoding" header field.
232             * 
233             * @return The content encoding of the response, or {@code null} if not 
234             *         known.
235             */
236            public String getContentEncoding() {
237    
238                    return contentEncoding;
239            }
240    }
