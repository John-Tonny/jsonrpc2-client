001    package com.thetransactioncompany.jsonrpc2.client;
002    
003    
004    /**
005     * Thrown to indicate a JSON-RPC 2.0 client session exception. Allows a general
006     * cause type to be specified to ease diagnostics and exception reporting.
007     *
008     * @author Vladimir Dzhuvinov
009     * @version 1.14 (2012-02-16)
010     */
011    public class JSONRPC2SessionException extends Exception {
012    
013            
014            /**
015             * Unspecified cause type.
016             */
017            public static final int UNSPECIFIED = 0;
018    
019            
020            /**
021             * The exception cause is network or I/O related.
022             */
023            public static final int NETWORK_EXCEPTION = 1;
024            
025            
026            /**
027             * Unexpected "Content-Type" header value of the HTTP response.
028             */
029            public static final int UNEXPECTED_CONTENT_TYPE = 2;
030            
031            
032            /**
033             * Invalid JSON-RPC 2.0 response (invalid JSON or invalid JSON-RPC 2.0
034             * response message).
035             */
036            public static final int BAD_RESPONSE = 3;
037            
038            
039            /**
040             * Unexpected JSON-RPC 2.0 response result (the result doesn't match the
041             * JSON type / format expected by the client).
042             */
043            public static final int UNEXPECTED_RESULT = 4;
044            
045            
046            /**
047             * Received a JSON-RPC 2.0 error response.
048             */
049            public static final int JSONRPC2_ERROR = 5;
050            
051            
052            /**
053             * Indicates the type of cause for this exception, see
054             * constants.
055             */
056            private int causeType;
057            
058            
059            /**
060             * Creates a new JSON-RPC 2.0 session exception with the specified 
061             * message; the cause type is {@link #UNSPECIFIED}.
062             *
063             * @param message The message.
064             */
065            public JSONRPC2SessionException(final String message) {
066            
067                    super(message);
068                    causeType = UNSPECIFIED;
069            }
070            
071            
072            /**
073             * Creates a new JSON-RPC 2.0 session exception with the specified 
074             * message and cause type.
075             *
076             * @param message   The message.
077             * @param causeType The cause type, see the constants.
078             */
079            public JSONRPC2SessionException(final String message, final int causeType) {
080            
081                    super(message);
082                    this.causeType = causeType;
083            }
084            
085            
086            /**
087             * Creates a new JSON-RPC 2.0 session exception with the specified 
088             * message, cause type and cause.
089             *
090             * @param message   The message.
091             * @param causeType The cause type, see the constants.
092             * @param cause     The original exception.
093             */
094            public JSONRPC2SessionException(final String message, final int causeType, final Throwable cause) {
095            
096                    super(message, cause);
097                    this.causeType = causeType;
098            }
099            
100            
101            /**
102             * Returns the exception cause type.
103             *
104             * @return The cause type constant.
105             */
106            public int getCauseType() {
107            
108                    return causeType;
109            }
110    }
