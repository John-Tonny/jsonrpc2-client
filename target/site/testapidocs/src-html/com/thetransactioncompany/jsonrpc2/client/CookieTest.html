015    /**
016     * Tests the cookie management.
017     *
018     * @author Vladimir Dzhuvinov
019     * @version $version$ (2013-01-18)
020     */
021    public class CookieTest extends TestCase {
022            
023            
024            public void testCookieSet() {
025    
026                    HttpCookie c1 = new HttpCookie("sessionid", "1000");
027                    HttpCookie c2 = new HttpCookie("sessionid", "2000");
028    
029                    System.out.println("Cookie 1 hash code: " + c1.hashCode());
030                    System.out.println("Cookie 2 hash code: " + c2.hashCode());
031    
032                    assertEquals(c1.hashCode(), c2.hashCode());
033    
034                    Set<HttpCookie> cookieSet = new HashSet<HttpCookie>();
035    
036                    cookieSet.add(c1);
037                    assertEquals(1, cookieSet.size());
038    
039                    cookieSet.add(c2);
040                    assertEquals(1, cookieSet.size());
041    
042                    for (HttpCookie c: cookieSet) {
043    
044                            assertEquals("1000", c.getValue());
045                    }
046            }
047    
048    
049            public void testSingleCookieOperation()
050                    throws Exception {
051                    
052                    CookieTestServer server = new CookieTestServer(18080);
053                    
054                    URL url = new URL("http://localhost:18080/jsonrpc2/");
055    
056                    JSONRPC2Session client = new JSONRPC2Session(url);
057                    
058                    client.getOptions().acceptCookies(true);
059                    
060                    JSONRPC2Request req = new JSONRPC2Request("test.cookie", 0);
061                    JSONRPC2Response resp = client.send(req);
062                    System.out.println(resp);
063                    
064                    
065                    System.out.println("Listing received cookies:");
066    
067                    List<HttpCookie> cookies = client.getCookies();
068    
069                    assertEquals(1, cookies.size());
070    
071                    for (HttpCookie cookie: cookies) {
072    
073                            System.out.println("\t" + cookie);
074    
075                            assertEquals("sessionid", cookie.getName());
076                            assertEquals("123", cookie.getValue());
077                    }
078                    
079                    // resend request
080                    resp = client.send(req);
081                    
082                    server.stop();
083            }
084    }
