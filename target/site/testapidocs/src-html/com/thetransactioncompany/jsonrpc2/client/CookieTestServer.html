013    /**
014     * Test JSON-RPC 2.0 server with cookies.
015     *
016     * @author Vladimir Dzhuvinov
017     * @version $version$ (2011-08-23)
018     */
019    public class CookieTestServer extends NanoHTTPD {
020            
021            
022            public CookieTestServer(final int port)
023                    throws IOException {
024                    
025                    super(port, null);
026            }
027            
028            
029            public NanoHTTPD.Response serve(final String uri, 
030                                            final String method, 
031                                            final Properties header, 
032                                            final Properties parms, 
033                                            final Properties files) {
034                    
035                    System.out.println("Cookie test server: Cookie: " + header.get("cookie"));
036                    
037                    JSONRPC2Response json = new JSONRPC2Response("Hello world!", 0);
038                    
039                    NanoHTTPD.Response out = new NanoHTTPD.Response(HTTP_OK, "application/json", json.toString());
040                    
041                    out.addHeader("Set-Cookie", "sessionid=123");
042                    
043                    return out;
044            }
045    }
